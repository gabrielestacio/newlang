fun int mdc(int n, int m) {
    if (m%n == 0) {
        return m;
    }else if (n%m == 0) {
        return n;
    }else if (n <= m) {
        return mdc(n, m % n);
    }else{
        return mdc(m, n % m);
    }

}

    int n = 0;
    int m = 0;
    int r = 1;

    read(n);
